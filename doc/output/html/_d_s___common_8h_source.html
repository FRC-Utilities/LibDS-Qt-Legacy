187  static inline QString getStaticIP (int net, int team, int host) {
188  QString string = QString ("%1").arg (team);
189  switch (string.length()) {
190  case 1:
191  string = "00.0" + string;
192  break;
193  case 2:
194  string = "00." + string;
195  break;
196  case 3:
197  string = QString ("0%1.%2%3").arg (string.at (0),
198  string.at (1),
199  string.at (2));
200  break;
201  case 4:
202  string = QString ("%1%2.%3%4").arg (string.at (0),
203  string.at (1),
204  string.at (2),
205  string.at (3));
206  break;
207  }
208  return QString ("%1.%2.%3").arg (QString::number (net),
209  string,
210  QString::number (host));
211  }

298  static inline QByteArray readSocket (QUdpSocket* socket) {
299  QByteArray data;

301  if (socket) {

303  data.resize (socket->pendingDatagramSize());
