213  static inline QString getStaticIP (int net, int team, int host) {
214  QString string = QString ("%1").arg (team);
215  switch (string.length()) {
216  case 1:
217  string = "00.0" + string;
218  break;
219  case 2:
220  string = "00." + string;
221  break;
222  case 3:
223  string = QString ("0%1.%2%3").arg (string.at (0),
224  string.at (1),
225  string.at (2));
226  break;
227  case 4:
228  string = QString ("%1%2.%3%4").arg (string.at (0),
229  string.at (1),
230  string.at (2),
231  string.at (3));
232  break;
233  }
234  return QString ("%1.%2.%3").arg (QString::number (net),
235  string,
236  QString::number (host));
237  }

324  static inline QByteArray readSocket (QUdpSocket* socket) {
325  QByteArray data;
326 
327  if (socket) {
328  while (socket->hasPendingDatagrams()) {
329  data.resize (socket->pendingDatagramSize());
330  socket->readDatagram (data.data(), data.size());
331  }
332  }
333 
334  return data;
335  }
